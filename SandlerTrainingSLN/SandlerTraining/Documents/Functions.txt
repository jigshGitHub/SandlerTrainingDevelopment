4)Function fnSetWebEdit(oDlgProperties,oEditProperties,Val2Set)

On error resume next
err.clear

Dialog(oDlgProperties).Winedit(oEditProperties).Set Val2Set

    If err.description<>"" Then
     Call RecoveryFunction(oDlgProperties,err.description)
     Call fnSetWebEdit (oDlgProperties,oEditProperties,Val2Set)
    End If

End Function

Function RecoveryFunction(oParentObjectProperties,ErrDescription)

   Select Case ErrDescription
     Case "Object is disabled"

     Set oDescription=Description.Create
     oDescription.Add "micclass","Dialog"

     Set objlist=Dialog(oParentObjectProperties).ChildObjects(oDescription)
     If objlist.count<>0 Then
      objlist(0).close
     End If
   End Select
   
End Function
